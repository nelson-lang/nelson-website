85  bool XmlDocListOfFiles::read()
86  {
87  bool haveChapterFile = false;
88  std::wstring chapterFilename = L"";
89  for (size_t k = 0; k < this->srcFiles.size(); k++)
90  {
91  bool b = boost::algorithm::ends_with(this->srcFiles[k], L"chapter.xml");
92  if (b)
93  {
94  if (chapterFilename != L"")
95  {
96  this->lastError = _W("multiple chapter.xml definition.");
97  return false;
98  }
99  chapterFilename = this->srcFiles[k];
100  haveChapterFile = true;
101  }
102  }
103  for (size_t k = 0; k < this->srcFiles.size(); k++)
104  {
105  bool b = boost::algorithm::ends_with(this->srcFiles[k], L"chapter.xml");
106  if (!b)
107  {
108  XmlDocDocument *xmlDoc = new XmlDocDocument(this->srcFiles[k], this->dstDirectory, this->bOverwriteExistingFiles, this->isQtHelp);
109  if (xmlDoc)
110  {
111  if (xmlDoc->readFile())
112  {
113  xmlItems.push_back(xmlDoc);
114  }
115  else
116  {
117  wstringVector errors = xmlDoc->getError();
118  this->lastError = _W("Error in file:") + L"\n" + this->srcFiles[k];
119  delete xmlDoc;
120  xmlDoc = nullptr;
121  if (!errors.empty())
122  {
123  this->lastError = this->lastError + L"\n" + errors[0];
124  }
125  this->clearItems();
126  return false;
127  }
128  }
129  else
130  {
131  this->lastError = ERROR_MEMORY_ALLOCATION;
132  this->clearItems();
133  return false;
134  }
135  }
136  }
137  if (!haveChapterFile)
138  {
139  this->lastError = _W("chapter.xml file is missing.");
140  this->clearItems();
141  return false;
142  }
143  if (haveChapterFile)
144  {
145  XmlDocDocument *xmlDoc = new XmlDocDocument(chapterFilename, this->dstDirectory, this->bOverwriteExistingFiles, this->isQtHelp);
146  if (xmlDoc)
147  {
148  if (xmlDoc->readFile())
149  {
150  if (xmlDoc->isChapterDocument())
151  {
152  this->moduleName = xmlDoc->getModuleName();
153  boost::container::vector<XmlDocGenericItem *> items = xmlDoc->getXmlDocGenericItems();
154  XmlDocChapterIndexItem *indexItem = new XmlDocChapterIndexItem();
155  for (size_t k = 0; k < xmlItems.size(); k++)
156  {
157  bool bSuccess;
158  indexItem->append(xmlItems[k]->getPageTitle(), RelativePath(dstDirectory, xmlItems[k]->getDestinationFile(), bSuccess), xmlItems[k]->getPageDescription());
159  }
160  items.push_back(indexItem);
161  std::wstring htmlFilenameDestination = L"";
162  if (this->moduleName != L"")
163  {
164  htmlFilenameDestination = dstDirectory + L"/chapter_" + this->moduleName + L".html";
165  this->chapterResultFilename = htmlFilenameDestination;
166  }
167  else
168  {
169  htmlFilenameDestination = this->chapterResultFilename;
170  }
171  XmlDocDocument *xmlChapterDoc = new XmlDocDocument(items, chapterFilename, htmlFilenameDestination, this->bOverwriteExistingFiles, this->isQtHelp);
172  xmlItems.push_back(xmlChapterDoc);
173  this->chapterTitle = xmlDoc->getChapter();
174  }
175  else
176  {
177  wstringVector errors = xmlDoc->getError();
178  this->lastError = _W("chapter.xml is missing.");
179  delete xmlDoc;
180  xmlDoc = nullptr;
181  this->clearItems();
182  return false;
183  }
184  }
185  else
186  {
187  wstringVector errors = xmlDoc->getError();
188  this->lastError = _W("Error in file:") + L"\n" + chapterFilename + L"\n";
189  delete xmlDoc;
190  xmlDoc = nullptr;
191  if (!errors.empty())
192  {
193  this->lastError = this->lastError + L"\n" + errors[0];
194  }
195  this->clearItems();
196  return false;
197  }
198  }
199  else
200  {
201  this->lastError = ERROR_MEMORY_ALLOCATION;
202  this->clearItems();
203  return false;
204  }
205  }
206  for (size_t k = 0; k < xmlItems.size(); k++)
207  {
208  if (k > 0)
209  {
210  std::wstring linkName = xmlItems[k - 1]->getPageTitle();
211  bool bRes;
212  std::wstring linkUrl = RelativePath(dstDirectory, xmlItems[k - 1]->getDestinationFile(), bRes);
213  xmlItems[k]->setPreviousPageLink(linkName, linkUrl);
214  }
215  if (!this->chapterTitle.empty())
216  {
217  bool bRes;
218  std::wstring linkUrl = L"";
219  boost::filesystem::path destPath(this->dstDirectory);
220  if (this->moduleName != L"")
221  {
222  destPath = destPath / (L"chapter_" + this->moduleName + L".html");
223  }
224  else
225  {
226  destPath = destPath / L"chapter.html";
227  }
228  linkUrl = RelativePath(dstDirectory, destPath.generic_wstring(), bRes);
229  xmlItems[k]->setIndexPageLink(this->chapterTitle, linkUrl);
230  }
231  if (k < xmlItems.size() - 1)
232  {
233  std::wstring linkName = xmlItems[k + 1]->getPageTitle();
234  bool bRes;
235  std::wstring linkUrl = RelativePath(dstDirectory, xmlItems[k + 1]->getDestinationFile(), bRes);
236  xmlItems[k]->setNextPageLink(linkName, linkUrl);
237  }
238  }
239  // last element is the chapter page
240  std::wstring linkName = xmlItems[0]->getPageTitle();
241  bool bRes;
242  std::wstring linkUrl = RelativePath(dstDirectory, xmlItems[0]->getDestinationFile(), bRes);
243  xmlItems[xmlItems.size() - 1]->setNextPageLink(linkName, linkUrl);
244  if (!sectionUpName.empty())
245  {
246  xmlItems[xmlItems.size() - 1]->setPreviousPageLink(sectionUpName, sectionUpUrl);
247  }
248  return true;
249  }
