37  bool deleteAst(ASTPtr pt, boost::container::vector<ASTPtr> v)
38  {
39  bool bFind = false;
40  boost::container::vector<ASTPtr>::iterator it = std::find(v.begin(), v.end(), pt);
41  if (it != v.end())
42  {
43  delete pt;
44  v.erase(it);
45  bFind = true;
46  }
47  return bFind;
48  }
