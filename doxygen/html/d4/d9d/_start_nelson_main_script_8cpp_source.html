26 bool StartNelsonMainScript(Evaluator* eval)
27 {
28  Context *ctx = eval->getContext();
29  if (ctx)
30  {
31  std::wstring rootPath = Nelson::GetRootPath();
32  boost::filesystem::path path(rootPath);
33  path += L"/etc/startup.nls";
34  bool bIsFile = boost::filesystem::exists(path) && !boost::filesystem::is_directory(path);
35  if (bIsFile)
36  {
37  std::wstring wstr = path.generic_wstring();
38  try
39  {
40  EvaluateScriptFile(eval, wstr.c_str());
41  }
42  catch (Exception &e)
43  {
44  // close all files in case of error at start-up
45  CloseAllFiles();
46  Interface *io = eval->getInterface();
47  e.what();
48  eval->setLastException(e);
49  std::wstring errmsg = _W("Main startup.nls failed to run.");
50  if (io)
51  {
52  io->errorMessage(errmsg);
53  }
54  else
55  {
56  errmsg = errmsg + L"\n";
57  fwprintf(stderr, errmsg.c_str());
58  }
59  }
60  return true;
61  }
62  return false;
63  }
64  return false;
65 }
